## Install

```bash
npm install --save common-utils
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'common-utils'
import 'common-utils/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
